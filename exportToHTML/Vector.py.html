from math import acos


class Vector:
    """A standard vector with components i, j, and k. If you want only xy vectors, set k to 0."""
    def __init__(self, i, j, k):
        self.i, self.j, self.k = i, j, k

    def __add__(self, other):
        """Adding another vector to your vector. You can use the + symbol."""
        return Vector(
            self.i + other.i,
            self.j + other.j,
            self.k + other.k
        )

    def __sub__(self, other):
        """Subtracting another vector from your vector. You can use the - symbol."""
        return Vector(
            self.i - other.i,
            self.j - other.j,
            self.k - other.k
        )

    def __mul__(self, other):
        """Multiplying a vector by a constant (scalar). Not two vectors!"""
        return Vector(
            self.i * other,
            self.j * other,
            self.k * other
        )

    def is_zero(self):
        """Is it the zero vector?"""
        return self.i == 0 and self.j == 0 and self.k == 0

    def to_string(self):
        return f"<{self.i}, {self.j}, {self.k}>"

    def cross(self, other):
        """Cross product of your vector and another vector."""
        new_i = self.j * other.k - self.k * other.j
        new_j = self.k * other.i - self.i * other.k
        new_k = self.i * other.j - self.j * other.i
        return Vector(new_i, new_j, new_k)

    def dot(self, other):
        """Dot product of your vector and another vector."""
        return self.i * other.i + self.j * other.j + self.k * other.k

    def abs(self):
        """Length"""
        return pow((pow(self.i, 2) + pow(self.j, 2) + pow(self.k, 2)), .5)

    def unit(self):
        """Returns a unit vector in the same direction as your vector."""
        return Vector(self.i / self.abs(), self.j / self.abs(), self.k / self.abs())

    def scalar_proj(self, other):
        """Returns the length of another vector projected onto yours."""
        dp = self.dot(other)
        if (self.abs() != 0):
            return dp / self.abs()
        else:
            return 0

    def vector_proj(self, other):
        """Returns another vector projected onto yours."""
        return self.unit() * self.scalar_proj(other)

    def angle_from(self, other):
        """Returns the angle between your vector and another vector."""
        return acos(self.dot(other) / (self.abs() * other.abs()))

b = Vector(10, 11, 12)
